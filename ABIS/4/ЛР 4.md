# -*- coding: utf-8 -*-
from odoo import http
from odoo.http import request
from datetime import date
import io

class TestModuleController(http.Controller):

    @http.route('/test_module/export_payments', type='http', auth='user')
    def export_payments(self, **kw):
        # Получаем всех сотрудников
        employees = request.env['hr.employee'].sudo().search([])

        output = io.StringIO()
        today = date.today()

        for emp in employees:
            # Получаем все действующие контракты на сегодня
            contracts = request.env['hr.contract'].sudo().search([
                ('employee_id', '=', emp.id),
                ('date_start', '<=', today),
                '|', ('date_end', '>=', today), ('date_end', '=', False)
            ])
            total_salary = sum(c.wage for c in contracts if c.wage)
            iban = emp.iban if hasattr(emp, 'iban') else ''
            fullname = f"{emp.name or ''}"
            output.write(f"{iban} {total_salary} {fullname}\n")

        response = request.make_response(
            output.getvalue(),
            headers=[
                ('Content-Type', 'text/plain; charset=utf-8'),
                ('Content-Disposition', 'attachment; filename=payments.txt')
            ]
        )
        return response
